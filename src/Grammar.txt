Expressions:

Expr        ::= AssignExpr
              | Expr ',' AssignExpr

AssignExpr  ::= BinaryExpr
              | UnaryExpr '=' AssignExpr

BinaryExpr  ::= UnaryExpr
              | BinaryExpr 'BOp' BinaryExpr

UnaryExpr   ::= 'UOp' UnaryExpr
              | 'sizedof' UnaryExpr
              | PostExpr

PostExpr    ::= PostExpr '[' Expr ']'
              | PostExpr '(' ')'
              | PostExpr '(' Expr ')'
              | PostExpr '.' identifier
              | PostExpr '->' identifier
              | PrimaryExpr

PrimaryExpr ::= identifer
              | constant
              | string-literal
              | '(' Expr ')'
